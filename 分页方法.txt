def page(r,data,pagenum,path,name,var):
	count=data.count()
	paginator = Paginator(data, pagenum)
	p = int(r.GET.get('p', 1))
	pagedata = paginator.page(p)
	pagecount = paginator.num_pages
	pagerange = paginator.page_range
	if p < 1:
		p = 1
	if p > pagecount:
		p = pagecount
	if p <= 5:
		page_list = pagerange[:10]
	elif p + 5 > pagecount:
		page_list = pagerange[-10:]
	else:
		page_list = pagerange[p - 5:p + 4]
	return render(r, path, {'pagedata': pagedata, 'page_list': page_list, 'p': p,'name':name,'count':count,'var':var})
